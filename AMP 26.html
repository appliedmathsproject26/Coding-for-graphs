import matplotlib.pyplot as plt
import numpy as np

# Data
iterations = [1, 2, 3]
energy_mj = [3.63, 4.651, 4.99842]

plt.figure()

# Plot line
plt.plot(iterations, energy_mj, marker='o', color='purple')

# Set x limit
plt.xlim(1, 3)

# Labels and title
plt.title("Energy lost vs External forces taken into account")
plt.xlabel("Iteration")
plt.ylabel("Total Energy done (MJ)")

# Set x-axis ticks every 1
plt.xticks(np.arange(1, 4, 1))

# Grid
plt.grid(True)

plt.show()

import matplotlib.pyplot as plt

# Data
mu = [0, 0.2, 0.4, 0.6, 0.8, 1, 1.2, 1.4, 1.6, 1.8, 2]

car1_energy = [2.55, 2.596, 2.654, 2.713, 2.77, 2.83, 2.889, 2.95, 3.0038, 3.062, 3.12]
car2_energy = [1.78, 1.84, 1.9, 1.962, 2.023, 2.084, 2.145, 2.205, 2.266, 2.327, 2.388]

# Plot
plt.figure()
plt.plot(mu, car1_energy, marker='o', color='royalblue', label='Car 1')
plt.plot(mu, car2_energy, marker='o', color='purple', label='Car 2')

# Labels and title
plt.xlabel('Coefficient of Friction (μ)')
plt.ylabel('Total Energy (MJ)')
plt.title('Coefficient of Friction vs Total Energy')

#force x-axis to start at 0 
plt.xlim(0, 2)
# Legend and grid
plt.legend()
plt.grid()

# Show graph
plt.show()

import matplotlib.pyplot as plt

# Time values
time = [0, 0.5, 1]

# Velocities
car1 = [80, 80, 40.7]
car2 = [72.5, 72.5, 51.91]

# Plot lines
plt.plot(time, car1, marker='o', color='royalblue', label='car 1')
plt.plot(time, car2, marker='o', color='purple', label='car2')

# Title and labels
plt.title("velocity vs time for collision")
plt.xlabel("time / s")
plt.ylabel("velocity / m/s")

# Show full y-axis range
plt.ylim(0, 100)
plt.yticks(range(0, 101, 25))  # 0,25,50,75,100

# Value labels
for x, y in zip(time, car1):
    plt.text(x, y + 1, str(y), ha='center')

for x, y in zip(time, car2):
    plt.text(x, y + 1, str(y), ha='center')

# Grid and legend
plt.grid(True)
plt.legend()

plt.show()
